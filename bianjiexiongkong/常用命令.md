
构建 chest_interfaces 功能包，它定义了本项目中使用的话题和服务消息类型
colcon build --packages-select chest_interfaces

构建并运行 button_simulator 功能包，该节点用于模拟物理按钮的按下事件
colcon build --packages-select button_simulator
source install/setup.bash
ros2 run button_simulator button_simulator_node
ros2 launch button_simulator button_simulator.launch.py

构建并运行 head_tracking 功能包，该节点负责处理头部追踪相关的算法与数据
colcon build --packages-select head_tracking
source install/setup.bash
ros2 run head_tracking head_tracking_node

构建并运行 video_source_loader 功能包，该节点根据配置文件加载并发布多个视频源
colcon build --packages-select video_source_loader
source install/setup.bash
ros2 run video_source_loader video_source_loader_node --ros-args -p config_file:=src/chest_interfaces/config/video_sources.yaml -p debug_output:=false

构建并运行 voice_processing 功能包，该节点负责处理语音输入（支持模拟）并发布识别结果
colcon build --packages-select voice_processing
source install/setup.bash
ros2 run voice_processing voice_processing_node --ros-args -p simulate_microphone:=true -p debug:=false

构建并运行 controller_display 功能包，该节点是控制系统的显示或UI界面
colcon build --packages-select controller_display
source install/setup.bash
ros2 run controller_display controller_display_node

构建并运行 ar_display 功能包，该节点负责处理AR效果并显示在屏幕上
colcon build --packages-select ar_display
source install/setup.bash
ros2 run ar_display ar_display_node

构建并运行 communication 功能包，该节点用于通讯
colcon build --packages-select communication
source install/setup.bash
ros2 launch communication communication.launch.py

构建并运行 web_monitor 功能包，该节点用于Web界面的启动
colcon build --packages-select web_monitor
source install/setup.bash
ros2 launch web_monitor web_monitor.launch.py


# 构建所有功能包
colcon build --packages-select chest_interfaces button_simulator head_tracking video_source_loader voice_processing controller_display ar_display

# 设置环境
source install/setup.bash

# 分别启动各个节点
ros2 launch button_simulator button_simulator.launch.py
ros2 launch head_tracking head_tracking.launch.py
ros2 launch video_source_loader video_source_loader.launch.py
ros2 launch voice_processing voice_processing.launch.py
ros2 launch controller_display controller_display.launch.py
ros2 launch ar_display ar_display.launch.py
